t = int(input())
for _ in range(t):
    a,b = map(int,input().split())
    xk,yk = map(int,input().split())
    xq,yq = map(int,input().split())
 
    set1 = set([(xk+a,yk+b),(xk+a,yk-b),(xk-a,yk+b),(xk-a,yk-b),(xk+b,yk+a),(xk+b,yk-a),(xk-b,yk+a),(xk-b,yk-a)])
    set2 = set([(xq+a,yq+b),(xq+a,yq-b),(xq-a,yq+b),(xq-a,yq-b),(xq+b,yq+a),(xq+b,yq-a),(xq-b,yq+a),(xq-b,yq-a)])
 
    print(len(set1&set2))
