n,k = map(int,input().split())
arr = list(map(int,input().split()))
 
Min = float('inf')
ans = None
 
i = 0
j = i+k-1
 
curr = sum(arr[:k])
 
while j<n:
    if curr<Min:
        ans = i
        Min = curr
    
    curr -= arr[i]
    
    i += 1
    j += 1
 
    if j<n:
        curr += arr[j]
 
print(ans+1)
