n = int(input())
arr = list(map(int,input().split()))
 
count = 0
ans = []
 
arr.sort()
 
small = arr[:n//2]
large = arr[n//2:]
 
for i in range(1,len(large)):
    if large[i-1]>small[i-1] and large[i]>small[i-1]:
        count+=1
    ans.append(large[i-1])
    ans.append(small[i-1])
 
ans.append(large[-1])
 
if n%2==0:
    ans.append(small[-1])
 
print(count)
print(*ans)
