t = int(input())
for _ in range(t):
    n = int(input())
    arr = list(map(int,input().split()))
 
    ans = []
 
    for i in range(n):
        g = 0
        l = 0
 
        for j in range(i+1,n):
            if arr[i]<arr[j]:
                g+=1
            elif arr[i]>arr[j]:
                l+=1
        
        ans.append(max(g,l))
    
    print(*ans)
