t = int(input())
for _ in range(t):
    n,c,k = map(int,input().split())
    arr = list(map(int,input().split()))
 
    arr.sort()
 
    for i in arr:
        if i<=c:
            x = c-i
 
            if x<=k:
                k-=x
                c+=i+x
            else:
                c+=i+k
                k = 0
        else:
            break
        
        # print(c)
    
    print(c)
