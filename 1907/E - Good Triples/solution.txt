ways = [0]*10
ways[0] = 1
 
for i in range(1,10):
    for a in range(i+1):
        for b in range(i+1):
            if a+b>i:
                continue
            
            ways[i] += 1
 
# print(ways)
 
t = int(input())
for _ in range(t):
    n = input()
 
    ans = 1
    
    for i in range(len(n)):
        ans *= ways[int(n[i])]
    
    print(ans)
