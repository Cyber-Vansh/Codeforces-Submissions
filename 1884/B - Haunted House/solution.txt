t = int(input())
for _ in range(t):
    n = int(input())
    s = input()
 
    ans = []
 
    zeros = []
    for i in range(n-1,-1,-1):
        if s[i]=="0":
            zeros.append(i)
    
    idx = 0
    curr = 0
 
    for i in range(n-1,-1,-1):
        if idx>=len(zeros):
            ans.append(-1)
        else:
            curr += i-zeros[idx]
            idx += 1
            ans.append(curr)
    
    print(*ans)
