n,m = map(int,input().split())
a = list(map(int,input().split()))
b = list(map(int,input().split()))
 
ans = float('inf')
 
left = 0
right = 3*(10**9)
 
while left<=right:
    mid = (left+right)//2
 
    i = 0
    j = 0
 
    while i<n and j<m:
        if abs(a[i]-b[j])<=mid:
            i += 1
        else:
            j += 1
        
        # print(mid,i,j)
        
    if i==n:
        ans = min(ans,mid)
        right = mid-1
    else:
        left = mid+1
 
print(ans)
