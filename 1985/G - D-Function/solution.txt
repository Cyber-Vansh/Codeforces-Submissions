mod = 10**9+7
 
t = int(input())
for _ in range(t):
    l,r,k = map(int,input().split())
 
    if k>9:
        print(0)
        continue
 
    x = None
 
    if k>=5:
        x = 1
    elif k==4:
        x = 2
    elif k==3:
        x = 3
    elif k==2:
        x = 4
    else:
        x = 9
    
    invx = pow(x,mod-2,mod)
    ans = (x * ((((pow(x+1,r,mod)-1) * invx) - (pow(x+1,l,mod)-1) * invx) % mod)) % mod
 
    print(ans)
