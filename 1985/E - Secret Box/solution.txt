t = int(input())
for _ in range(t):
    x,y,z,k = map(int,input().split())
 
    ans = 0
 
    for a in range(1,x+1):
        for b in range(1,y+1):
            if a*b>k:
                break
            
            # print(a,b,k)
                
            if k%(a*b)==0:
                c = k//(a*b)
            
                ans = max(ans,(x-a+1)*(y-b+1)*(z-c+1))
    
    print(ans)
