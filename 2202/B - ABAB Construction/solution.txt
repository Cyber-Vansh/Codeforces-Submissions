t = int(input())
for _ in range(t):
    n = int(input())
    x = input()
 
    check = True
 
    if n%2==1:
        if x[0]=="b":
            check = False
 
        for i in range(2,n,2):
            if x[i]!="?" and x[i-1]!="?" and x[i]==x[i-1]:
                check = False
                break
    else:
        for i in range(1,n,2):
            if x[i]!="?" and x[i-1]!="?" and x[i]==x[i-1]:
                check = False
                break
 
    print("YES" if check else "NO")
