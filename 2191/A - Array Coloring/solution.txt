t = int(input())
 
for _ in range(t):
    n = int(input())
    a = list(map(int,input().split()))
 
    pos = [0]*(n+1)
    for i,v in enumerate(a):
        pos[v] = i
 
    ok = True
    for x in range(1,n):
        if abs(pos[x]-pos[x+1])%2==0:
            ok = False
            break
 
    print("YES" if ok else "NO")
