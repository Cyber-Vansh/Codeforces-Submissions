t = int(input())
for _ in range(t):
    n,k = map(int,input().split())
 
    ans = 0
 
    for i in range(20):
        cost = 1<<i 
 
        if n//cost>=k:
            ans += k
            n -= cost*k
        else:
            x = n//cost
 
            ans += x
            n -= cost*x
    
    print(ans)
