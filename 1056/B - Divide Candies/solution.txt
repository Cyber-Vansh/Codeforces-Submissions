n,m = map(int,input().split())
 
mods = {}
 
for i in range(1,m+1):
    mods[(i**2)%m] = mods.get((i**2)%m,0)+n//m
 
for i in range(m*(n//m)+1,n+1):
    mods[(i**2)%m] = mods.get((i**2)%m,0)+1
 
ans = mods.get(0,0)*mods.get(0,0)
 
for i in range(1,m):
    x = m-i
 
    if x<=i:
        if x==i:
            ans += mods.get(x,0)*mods.get(x,0)
        break
    
    ans += (mods.get(i,0)*mods.get(x,0))*2
 
print(ans)
