t = int(input())
for _ in range(t):
    w,h,a,b = map(int,input().split())
    x1,y1,x2,y2 = map(int,input().split())
 
    def canTileRoof(w, h, a, b, x1, y1, x2, y2):
        if (abs(x1-x2)%a==0 and x1!=x2) or (abs(y1-y2)%b==0 and y1!=y2):
            return "Yes"
        else:
            return "No"
    print(canTileRoof(w, h, a, b, x1, y1, x2, y2))
