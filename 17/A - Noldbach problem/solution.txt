n,k = map(int,input().split())
 
arr = [True]*(n+1)
arr[0] = False
arr[1] = False
 
primes = []
 
for i in range(2,len(arr)):
    if arr[i]:
        primes.append(i)
        for j in range(i*i,len(arr),i):
            arr[j] = False
 
total = 0
 
for i in range(1,len(primes)):
    if primes[i-1]+primes[i]+1<=n and arr[primes[i-1]+primes[i]+1]:
        total+=1
 
print("YES" if total>=k else "NO")
