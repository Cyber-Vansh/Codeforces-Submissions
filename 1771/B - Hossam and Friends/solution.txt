t = int(input())
for _ in range(t):
    n,m = map(int,input().split())
 
    not_friends = [n+1]*(n+1)
 
    for i in range(m):
        a,b = map(int,input().split())
 
        if b>a:
            not_friends[a] = min(b,not_friends[a])
        else:
            not_friends[b] = min(a,not_friends[b])
    
    ans = 0
 
    for i in range(n-1,0,-1):
        not_friends[i-1] = min(not_friends[i-1],not_friends[i])
    
    for i in range(1,n+1):
        ans += not_friends[i]-i
    
    print(ans)
