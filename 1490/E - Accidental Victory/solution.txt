t = int(input())
for _ in range(t):
    n = int(input())
    arr = list(map(int,input().split()))
 
    arr = [(arr[i],i+1) for i in range(n)]
    
    arr.sort()
 
    pre = [0]*n
    pre[0] = arr[0][0]
    for i in range(1,n):
        pre[i] = pre[i-1]+arr[i][0]
    
    valid = [arr[-1][1]]
 
    for i in range(n-2,-1,-1):
        if pre[i]>=arr[i+1][0]:
            valid.append(arr[i][1])
        else:
            break
    
    print(len(valid))
    print(*sorted(valid))
