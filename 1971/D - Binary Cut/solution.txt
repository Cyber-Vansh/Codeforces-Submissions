t = int(input())
for _ in range(t):
    s = input()
 
    flag = False
 
    ans = 1
    prev = s[0]
 
    for i in range(1,len(s)):
        if s[i]==prev:
            continue
        
        ans += 1
 
        if prev=="0":
            flag = True
        
        prev = s[i]
    
    if flag:
        ans -= 1
    
    print(ans)
