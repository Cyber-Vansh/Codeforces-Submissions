t = int(input())
for _ in range(t):
    n = int(input())
    s = input()
 
    ones = s.count("1")
 
    if ones==0:
        print((n+2)//3)
        continue
 
    parts = s.split("1")
    add = 0
 
    add += (len(parts[0])+1)//3
 
    for mid in parts[1:-1]:
        add += len(mid)//3
 
    add += (len(parts[-1])+1)//3
 
    print(ones+add)
