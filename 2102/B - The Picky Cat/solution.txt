t = int(input())
for _ in range(t):
    n = int(input())
    arr = list(map(int,input().split()))
 
    small = 0
    big = 0
    
    for i in range(1,n):
        if abs(arr[0])>abs(arr[i]):
            small += 1
        else:
            big += 1
    
    if small<n//2 or (n%2==0 and big>=n//2-1) or (n%2==1 and big>=n//2):
        print("YES")
    else:
        print("NO")
