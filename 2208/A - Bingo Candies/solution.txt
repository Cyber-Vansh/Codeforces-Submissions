t = int(input())
for _ in range(t):
    n = int(input())
 
    mat = []
    for i in range(n):
        arr = list(map(int,input().split()))
        mat.append(arr)
    
    freq = {}
 
    for i in range(n):
        for j in range(n):
            freq[mat[i][j]] = freq.get(mat[i][j],0)+1
    
    check = True
 
    for i in freq.values():
        if i>n*(n-1):
            check = False
            break
    
    if check:
        print("Yes")
    else:
        print("No")
