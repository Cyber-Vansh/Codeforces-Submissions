t = int(input())
for _ in range(t):
    n = int(input())
    arr = list(map(int,input().split()))
 
    arr.sort(reverse=True)
 
    check = False
 
    for i in range(1,n):
        if arr[i]==arr[i-1]:
            check = True
            break
    
    if check:
        print(-1)
    else:
        print(*arr)
