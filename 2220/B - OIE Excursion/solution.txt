t = int(input())
for _ in range(t):
    n,m = map(int,input().split())
    arr = list(map(int,input().split()))
 
    curr = 1
    check = True
 
    for i in range(1, n):
        if arr[i]==arr[i-1]:
            curr += 1
            if curr>=m:
                check = False
                break
        else:
            curr = 1
 
    if check:
        print("YES")
    else:
        print("NO")
