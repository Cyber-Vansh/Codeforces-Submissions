fact = [1]
for i in range(1,11):
    fact.append(fact[-1]*i)
 
s1 = input()
s2 = input()
 
plus1 = s1.count("+")
minus1 = s1.count("-")
 
plus2 = s2.count("+")
minus2 = s2.count("-")
empty = s2.count("?")
 
if plus2>plus1 or minus2>minus1 :
    print(0)
else:
    need_plus = plus1-plus2
    need_minus = minus1-minus2
 
    num = fact[need_plus+need_minus]
    denom = fact[need_plus]*fact[need_minus]*(1<<empty)
 
    print(num/denom)
