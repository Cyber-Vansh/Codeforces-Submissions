import sys
input = sys.stdin.readline
 
t = int(input())
for _ in range(t):
    x = int(input())
 
    ans = []
        
    while x>0:
        if x%2==0:
            ans.append(0)
            x //= 2
        else:
            if x%4==1:
                ans.append(1)
                x = (x-1)//2
            else:
                ans.append(-1)
                x = (x+1)//2
    
    print(len(ans))
    print(*ans)
