t = int(input())
for _ in range(t):
    m = int(input())
    a = list(map(int,input().split()))
    b = list(map(int,input().split()))
 
    pre = [0]*m
    suff = [0]*m
    
    for i in range(1,m):
        pre[i] = pre[i-1]+b[i-1]
    
    for i in range(m-2,-1,-1):
        suff[i] = suff[i+1]+a[i+1]
 
    ans = float('inf')
 
    for i in range(m):
        ans = min(ans,max(pre[i],suff[i]))
    
    print(ans)
