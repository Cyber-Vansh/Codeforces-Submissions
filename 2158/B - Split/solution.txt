t = int(input())
for _ in range(t):
    n = int(input())
    arr = list(map(int,input().split()))
 
    odd = 0
    even = 0
 
    freq = {}
 
    for i in range(2*n):
        freq[arr[i]] = freq.get(arr[i],0)+1
    
    for i,j in freq.items():
        if j%2==0:
            even += 1
        else:
            odd += 1
    
    if odd>0:
        ans = odd+2*even
    else:
        if even%2==n%2:
            ans = 2*even
        else:
            ans = 2*even-2
    
    print(ans)
