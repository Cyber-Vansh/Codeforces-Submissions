t = int(input())
for _ in range(t):
    n = int(input())
    arr = list(map(int,input().split()))
 
    x = arr[0]
    ans = float('inf')
    curr = 0
 
    for i in range(n):
        if arr[i]==x:
            curr += 1
        else:
            ans = min(ans,curr)
            curr = 0
    ans = min(ans,curr)
 
    print(ans if ans!=n else -1)
