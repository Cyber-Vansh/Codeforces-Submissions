t = int(input())
for _ in range(t):
    s = input()
 
    count = 0
 
    i = 0
 
    while i<len(s):
        if i==0:
            if s[i]=="u":
                count+=1
            i+=1
        else:
            if s[i]=="u":
                if i+1<len(s):
                    if s[i+1]=="u":
                        count+=1
                    i+=2
                else:
                    count+=1
                    i+=1
            else:
                i+=1
    
    print(count)
