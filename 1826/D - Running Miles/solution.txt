t = int(input())
for _ in range(t):
    n = int(input())
    arr = list(map(int,input().split()))
 
    pre = [arr[0]]
    for i in range(1,n):
        pre.append(max(pre[-1]-1,arr[i]))
    
    suff = [arr[n-1]]
    for i in range(n-2,-1,-1):
        suff.append(max(suff[-1]-1,arr[i]))
    suff.reverse()
    
    ans = 0
 
    for i in range(1,n-1):
        ans = max(ans,arr[i]+pre[i-1]+suff[i+1])
    
    print(ans-2)
