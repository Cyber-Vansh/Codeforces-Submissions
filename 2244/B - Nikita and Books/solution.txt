import math
 
t = int(input())
for _ in range(t):
    n = int(input())
    arr = list(map(int,input().split()))
 
    check = True
 
    for i in range(n):
        x = arr[i]-(i+1)
 
        if x>0 and i<n-1:
            arr[i+1] += x
            arr[i] -= x
        
        if i>0 and arr[i]<=arr[i-1]:
            check = False
    
    if check:
        print("YES")
    else:
        print("NO")
