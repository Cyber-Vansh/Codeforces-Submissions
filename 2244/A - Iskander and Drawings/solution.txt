import math
 
t = int(input())
for _ in range(t):
    n = int(input())
    s = input()
 
    Max = 0
 
    count = 0
    for i in range(n):
        if s[i]=="#":
            count += 1
        else:
            Max = max(Max,count)
            count = 0
    Max = max(Max,count)
 
    print(math.ceil(Max/2))
