t = int(input())
for _ in range(t):
    n,q = map(int,input().split())
    s = input()
    
    pre = [0]*n
    
    for i in range(1,n):
        if s[i]==s[i-1]:
            pre[i] = 1
        pre[i] += pre[i-1]
    
    for i in range(q):
        l,r,k = map(int,input().split())
        
        if l==r:
            print("YES")
        else:
            if pre[r-1]-pre[l-1]<=2*k:
                print("YES")
            else:
                print("NO")
