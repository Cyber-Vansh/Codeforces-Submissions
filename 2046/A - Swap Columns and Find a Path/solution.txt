t = int(input())
for _ in range(t):
    n = int(input())
    a = list(map(int,input().split()))
    b = list(map(int,input().split()))
 
    ans = -float("inf")
 
    # for i in range(n):
    #     if a[i]+b[i]>ans:
    #         ans = a[i]+b[i]
    #         idx = i
 
    for j in range(n):
        curr = a[j]+b[j]
 
        for i in range(n):
            if i==j:
                continue
            
            curr += max(a[i],b[i])
 
        ans = max(ans,curr)
    
    print(ans)
