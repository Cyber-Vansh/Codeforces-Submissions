arr = [0]*5000001
spf = [0]*5000001
 
for i in range(2,len(arr)):
    if arr[i]==0: 
        spf[i]=i
        for j in range(i*i,len(arr),i): 
            arr[j]=1 
            if spf[j]==0: 
                spf[j]=i
 
for i in range(2,len(arr)): 
    arr[i] = arr[i//spf[i]] + 1
# print(arr)
 
for i in range(1,len(arr)):
    arr[i] = arr[i-1]+arr[i]
 
t = int(input())
for _ in range(t):
    a,b = map(int,input().split())
 
    print(arr[a]-arr[b])
