t = int(input())
for _ in range(t):
    n = int(input())
    s = input()
 
    ones = s.count("1")
 
    if n%2==1 and ones%2==1:
        print(-1)
        continue
 
    ans = []
 
    if ones%2==0:
        for i in range(n):
            if s[i]=="1":
                ans.append(i+1)
    else:
        for i in range(n):
            if s[i]=="0":
                ans.append(i+1)
 
    print(len(ans))
    if len(ans)>0:
        print(*ans)
